cmake_minimum_required(VERSION 3.12)
project(n_body_headless)

set(CMAKE_CXX_STANDARD 14)

set(CMAKE_CXX_FLAGS "${CMAKE_CXX_FLAGS} ${OpenMP_CXX_FLAGS} -foffload=-lm")

add_executable(n_body_headless main.cpp simulation.h simulation.cpp body.h body.cpp)

target_link_libraries(${CMAKE_PROJECT_NAME} ${OPENGL_LIBRARIES} ${GLUT_LIBRARY} )
